public class Triangle {
   private double side1, side2, side3;
   
   public Triangle(double side1, double side2, double side3)
           throws IllegalTriangleException {
       
       if (side1 <= 0 || side2 <= 0 || side3 <= 0) {
           throw new IllegalTriangleException("Все стороны должны быть положительными");
       }
       
       if (side1 + side2 <= side3 ||
           side1 + side3 <= side2 ||
           side2 + side3 <= side1) {
           throw new IllegalTriangleException(
               "Сумма любых двух сторон должна быть больше третьей стороны");
       }
       
       this.side1 = side1;
       this.side2 = side2;
       this.side3 = side3;
   }
   
   // Геттеры
   public double getSide1() { return side1; }
   public double getSide2() { return side2; }
   public double getSide3() { return side3; }
}

public class DuplicateRemover {
   public static <T> ArrayList<T> removeDuplicates(ArrayList<T> list) {
       ArrayList<T> result = new ArrayList<>();
       
       for (T item : list) {
           if (!result.contains(item)) {
               result.add(item);
           }
       }
       
       return result;
   }
   
   public static void main(String[] args) {
       ArrayList<String> list = new ArrayList<>();
       list.add("apple");

   public static int linearSearch(int[] array, int target) {
       for (int i = 0; i < array.length; i++) {
           if (array[i] == target) {

           }

   public Circle(double radius) {
       this.radius = radius;

       return radius;
   }
   
   public double getArea() {

       return Double.compare(this.radius, other.radius);

   @Override

       return String.format("Circle(radius=%.2f)", radius);

public class CircleFinder {
   public static Circle findLargestCircle(Circle[] circles) {
       if (circles == null || circles.length == 0) {
           return null;
       }
       
       Circle largest = circles[0];
       for (int i = 1; i < circles.length; i++) {
           if (circles[i].compareTo(largest) > 0) {
               largest = circles[i];
           }
       }
       
       return largest;
   }
   
   public static void main(String[] args) {
       Circle[] circles = {
           new Circle(5.0),
           new Circle(3.0),
           new Circle(7.0),
           new Circle(4.5)
       };
       
       Circle largest = findLargestCircle(circles);
       System.out.println("Самый большой круг: " + largest);
       System.out.println("Радиус: " + largest.getRadius());

   }

public class MaxIn2DArray {
   public static <T extends Comparable<T>> T findMax(T[][] matrix) {
       if (matrix == null || matrix.length == 0) {
           return null;
       }
       
       T max = matrix[0][0];
       for (T[] row : matrix) {
           for (T element : row) {
               if (element.compareTo(max) > 0) {
                   max = element;
               }
           }
       }
       return max;
   }
   
   public static void main(String[] args) {
       Integer[][] intMatrix = {
           {1, 5, 3},
           {9, 2, 8},
           {4, 7, 6}
       };
       
       Double[][] doubleMatrix = {
           {1.5, 3.7},
           {2.8, 4.1},
           {0.9, 5.3}
       };
       
       String[][] stringMatrix = {
           {"apple", "banana"},
           {"orange", "grape"}
       };
       
       System.out.println("Максимум в intMatrix: " + findMax(intMatrix));
       System.out.println("Максимум в doubleMatrix: " + findMax(doubleMatrix));

import java.util.Arrays;
 
public class GenericSorter {
   public static <T extends Comparable<T>> void sortArray(T[] array) {
       for (int i = 0; i < array.length - 1; i++) {
           for (int j = 0; j < array.length - i - 1; j++) {
               if (array[j].compareTo(array[j + 1]) > 0) {
                   T temp = array[j];
                   array[j] = array[j + 1];
                   array[j + 1] = temp;
               }
           }
       }
   }
   
   public static void main(String[] args) {
       // Сортировка массива целых чисел
       Integer[] intArray = {5, 2, 8, 1, 9};
       System.out.println("До сортировки (int): " + Arrays.toString(intArray));
       sortArray(intArray);
       System.out.println("После сортировки (int): " + Arrays.toString(intArray));
       
       // Сортировка массива double
       Double[] doubleArray = {3.5, 1.2, 4.8, 2.1};
       System.out.println("\nДо сортировки (double): " + Arrays.toString(doubleArray));
       sortArray(doubleArray);
       System.out.println("После сортировки (double): " + Arrays.toString(doubleArray));
       
       // Сортировка массива символов
       Character[] charArray = {'d', 'a', 'c', 'b'};
       System.out.println("\nДо сортировки (char): " + Arrays.toString(charArray));

import java.util.Arrays;
 
public class Student {
   private String name;
   private String surname;
   private int[] grades;
   private int gradeCount;
   
   public Student(String name, String surname) {
       this.name = name;
       this.surname = surname;
       this.grades = new int[10];
       this.gradeCount = 0;
   }
   
   // Геттеры и сеттеры для name
   public String getName() {
       return name;
   }
   
   public void setName(String name) {
       this.name = name;
   }
   
   // Геттеры и сеттеры для surname
   public String getSurname() {
       return surname;
   }
   
   public void setSurname(String surname) {
       this.surname = surname;
   }
   
   // Геттеры и сеттеры для grades
   public int[] getGrades() {
       return Arrays.copyOf(grades, gradeCount);
   }
   
   public void setGrades(int[] newGrades) {
       if (newGrades.length <= 10) {
           this.grades = Arrays.copyOf(newGrades, 10);
           this.gradeCount = newGrades.length;
       }
   }
   
   // Добавление новой оценки
   public void addGrade(int grade) {
       if (gradeCount < 10) {
           grades[gradeCount] = grade;
           gradeCount++;
       } else {
           // Сдвигаем оценки влево
           for (int i = 0; i < 9; i++) {
               grades[i] = grades[i + 1];
           }
           grades[9] = grade;
       }
   }
   
   // Расчет среднего балла
   public double getAverageGrade() {
       if (gradeCount == 0) {
           return 0.0;
       }
       
       int sum = 0;
       for (int i = 0; i < gradeCount; i++) {
           sum += grades[i];
       }
       return (double) sum / gradeCount;
   }
   
   @Override
   public String toString() {
       return String.format("%s %s, Средний балл: %.2f, Оценки: %s",
           name, surname, getAverageGrade(),
           Arrays.toString(getGrades()));
   }
   
   public static void main(String[] args) {
       Student student = new Student("Иван", "Иванов");
       
       student.addGrade(5);
       student.addGrade(4);
       student.addGrade(5);
       student.addGrade(3);
       
       System.out.println(student);
       
       // Добавляем еще оценки (больше 10)
       for (int i = 0; i < 15; i++) {
           student.addGrade(4);
       }
       
       System.out.println("После добавления 15 оценок:");
       System.out.println(student);
   }
}

class StudentService {

   public Student bestStudent(Student[] students) {
       if (students == null || students.length == 0) {
           return null;
       }
       
       Student best = students[0];
       for (Student student : students) {
           if (student.getAverageGrade() > best.getAverageGrade()) {
               best = student;
           }
       }
       return best;
   }
   
   // Сортирует студентов по фамилии
   public void sortBySurname(Student[] students) {
       Arrays.sort(students, new Comparator<Student>() {
           @Override
           public int compare(Student s1, Student s2) {
               return s1.getSurname().compareToIgnoreCase(s2.getSurname());
           }
       });
   }
}

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.List;
 
public class FileAnalyzer {
   public static void analyzeFile(String filePath) {
       try {
           Path path = Paths.get(filePath);
           List<String> lines = Files.readAllLines(path);
           
           // Подсчет строк
           int lineCount = lines.size();
           System.out.println("Общее количество строк: " + lineCount);
           
           // Поиск самых длинных строк
           int maxLength = 0;
           for (String line : lines) {
               if (line.length() > maxLength) {
                   maxLength = line.length();
               }
           }
           
           System.out.println("Максимальная длина строки: " + maxLength);
           System.out.println("Самые длинные строки:");
           
           for (String line : lines) {
               if (line.length() == maxLength) {
                   System.out.println(line);
               }
           }
           
       } catch (IOException e) {
           System.err.println("Ошибка при чтении файла: " + e.getMessage());
       }
   }
   
   public static void main(String[] args) {
       if (args.length != 1) {
           System.out.println("Использование: java FileAnalyzer <путь_к_файлу>");
           System.out.println("Пример: java FileAnalyzer test.txt");
           return;
       }
       
       analyzeFile(args[0]);
   }
}
17. Объединение двух файлов
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;
import java.util.List;
 
public class FileMerger {
   public static void mergeFiles(String file1Path, String file2Path, String outputPath) {
       try {
           // Чтение первого файла
           Path path1 = Paths.get(file1Path);
           List<String> content1 = Files.readAllLines(path1);
           
           // Чтение второго файла
           Path path2 = Paths.get(file2Path);
           List<String> content2 = Files.readAllLines(path2);
           
           // Создание/очистка выходного файла
           Path output = Paths.get(outputPath);
           Files.deleteIfExists(output);
           Files.createFile(output);
           
           // Запись содержимого первого файла
           Files.write(output, content1, StandardOpenOption.WRITE);
           
           // Добавление содержимого второго файла
           Files.write(output, content2, StandardOpenOption.APPEND);
           
           System.out.println("Файлы успешно объединены в: " + outputPath);
           
       } catch (IOException e) {
           System.err.println("Ошибка: " + e.getMessage());
       }
   }
   
   public static void main(String[] args) {
       if (args.length != 3) {
           System.out.println("Использование: java FileMerger <файл1> <файл2> <выходной_файл>");
           System.out.println("Пример: java FileMerger file1.txt file2.txt merged.txt");
           return;
       }
       
       mergeFiles(args[0], args[1], args[2]);
   }
}
